const SHEET_REGISTROS = 'Ingresos'; const SHEET_EMPLEADOS = 'Empleados'; function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheetReg = ss.getSheetByName(SHEET_REGISTROS); const dataReg = sheetReg.getDataRange().getValues(); const headers = dataReg[0]; const registros = dataReg.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); const sheetEmp = ss.getSheetByName(SHEET_EMPLEADOS); let empleados = []; if (sheetEmp) { const dataEmp = sheetEmp.getDataRange().getValues(); empleados = dataEmp.flat().filter(v => v && String(v).trim() !== ''); } const payload = JSON.stringify({ empleados, registros }); return ContentService .createTextOutput(payload) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const body = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); if (body.accion === 'guardarEmpleados') { let sheetEmp = ss.getSheetByName(SHEET_EMPLEADOS); if (!sheetEmp) sheetEmp = ss.insertSheet(SHEET_EMPLEADOS); sheetEmp.clearContents(); const nombres = (body.empleados || []).filter(n => n && String(n).trim()); if (nombres.length) { sheetEmp.getRange(1, 1, nombres.length, 1).setValues(nombres.map(n => [n])); } return ContentService .createTextOutput(JSON.stringify({ ok: true })) .setMimeType(ContentService.MimeType.JSON); } const sheet = ss.getSheetByName(SHEET_REGISTROS); sheet.appendRow([ body.timestamp, body.nombre, body.hora, body.tarde ]); return ContentService .createTextOutput(JSON.stringify({ ok: true })) .setMimeType(ContentService.MimeType.JSON); }
